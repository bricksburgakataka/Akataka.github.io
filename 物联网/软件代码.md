// ==================== 初始化WiFi ====================
void init_wifi(const char *ssid, const char *password) {
  WiFi.mode(WIFI_STA);
  WiFi.begin(ssid, password);
  Serial.print("📡 连接WiFi");
  
  int attempts = 0;
  while (WiFi.status() != WL_CONNECTED && attempts < 20) {
    delay(500);
    Serial.print(".");
    attempts++;
  }
  
  if (WiFi.status() == WL_CONNECTED) {
    Serial.println("\n✅ WiFi连接成功");
    Serial.print("📱 IP地址: ");
    Serial.println(WiFi.localIP());

  }
}

// ==================== MQTT回调函数 ====================
void mqtt_callback(char *topic, byte *payload, unsigned int length) {
  Serial.print("📨 收到消息 [");
  Serial.print(topic);
  Serial.print("] ");
  
  payload[length] = '\0';
  String message = String((char*)payload);
  Serial.println(message);

  if (strstr(topic, "sys/messages")) {
    handleAIResultMessage(message);
  }
  else if (strstr(topic, "sys/commands")) {
    handleCommandMessage(message, topic);
  }
  
  messageCount++;
}

// ==================== 处理AI结果消息 ====================
void handleAIResultMessage(String message) {
  Serial.println("\n🤖 === AI识别结果 ===");
  
  DynamicJsonDocument doc(2048);
  DeserializationError error = deserializeJson(doc, message);
  
  String result = "未知";
  
  if (!error) {
    if (doc.containsKey("content")) {
      String content = doc["content"];
      DynamicJsonDocument contentDoc(2048);
      DeserializationError contentError = deserializeJson(contentDoc, content);
      
      if (!contentError && contentDoc.containsKey("AIResult")) {
        result = contentDoc["AIResult"].as<String>();
      } else {
        result = content;
      }
    } 
    else if (doc.containsKey("AIResult")) {
      result = doc["AIResult"].as<String>();
    }
    else {
      result = message;
    }
  } else {
    result = message;
  }
  
  // 存储AI结果
  aiResult = result;
  aiResultType = "message";
  aiTimestamp = millis();
  hasAIResult = true;
  
  aiDetectionCount++;
  lastActivity = "AI识别: " + result.substring(0, 20) + "...";
  
  // ✅ 收到AI结果后立即单独上报AIResult
  Serial.println("🚀 收到AI结果，立即单独上报AIResult...");
  reportAIResultOnly();
  
  Serial.println("=====================\n");
}

// ==================== 处理命令消息 ====================
void handleCommandMessage(String message, char* topic) {
  DynamicJsonDocument doc(1024);
  DeserializationError error = deserializeJson(doc, message);
  
  if (error) {
    Serial.println("❌ JSON解析失败");
    errorCount++;
    return;
  }
  
  String commandName = doc["command_name"].as<String>();
  Serial.printf("🎯 收到命令: %s\n", commandName.c_str());
  
  if (commandName == "AIResult") {
    handleAIResultCommand(doc["paras"]);
    sendCommandResponse(topic);
  }
  else if (commandName == "WeatherUpdate") {
    handleWeatherCommand(doc["paras"]);
    sendCommandResponse(topic);
  }
  else if (commandName == "SystemControl") {
    handleSystemControlCommand(doc["paras"]);
    sendCommandResponse(topic);
  }
  else if (commandName == "TakePhoto") {
    handleTakePhotoCommand(doc["paras"], topic);
  }
  else if (commandName == "ScheduleUpdate") {
    handleScheduleCommand(doc["paras"]);
    sendCommandResponse(topic);
  }
  else {
    Serial.printf("⚠️ 未知命令类型: %s\n", commandName.c_str());
    sendCommandResponse(topic);
  }
  
  commandCount++;
  lastActivity = "执行命令: " + commandName;
}

// ==================== 立即发送命令响应 ====================
void sendCommandResponseImmediately(char* topic) {
  char requestId[100] = {0};
  char* pstr = topic;
  char* p = requestId;
  int flag = 0;
  
  while (*pstr) {
    if (flag) *p++ = *pstr;
    if (*pstr == '=') flag = 1;
    pstr++;
  }
  *p = '\0';
  
  char responseTopic[200] = {0};
  strcat(responseTopic, MQTT_TOPIC_CMD_RESPONSE);
  strcat(responseTopic, requestId);
  
  const char* responseData = "{\"result_code\": 0,\"response_name\": \"COMMAND_RESPONSE\",\"paras\": {\"result\": \"success\"}}";
  
  if (mqttClient.publish(responseTopic, responseData)) {
    Serial.println("✅ 命令响应已立即发送");
  } else {
    Serial.println("❌ 命令响应发送失败");
    errorCount++;
  }
}

// ==================== 核心函数：拍照并上传到云平台进行AI识别 ====================
void captureAndSendForAI() {
  isCapturing = true;
  Serial.println("\n🚀 开始AI识别流程...");
  
  camera_fb_t *fb = esp_camera_fb_get();
  if (!fb) {
    Serial.println("❌ 摄像头捕获失败");
    errorCount++;
    isCapturing = false;
    return;
  }
  
  Serial.printf("📷 获取到图片，大小: %d bytes\n", fb->len);
  photoCount++;
  
  String imageBase64 = base64::encode(fb->buf, fb->len);
  int totalLength = imageBase64.length();
  int totalChunks = (totalLength + BASE64_CHUNK_SIZE - 1) / BASE64_CHUNK_SIZE;
  
  Serial.printf("📦 Base64编码完成，长度: %d, 分片数: %d\n", totalLength, totalChunks);
  
  // 1. 发送开始标记
  bool startSuccess = reportImageStart(totalChunks, fb->len);
  mqttClient.loop();
  delay(200);
  
  if (!startSuccess) {
    Serial.println("❌ 开始标记发送失败，中止流程");
    esp_camera_fb_return(fb);
    isCapturing = false;
    return;
  }
  
  // 2. 分片发送图片数据
  for (int i = 0; i < totalChunks; i++) {
    if (!mqttClient.connected()) {
      Serial.println("❌ MQTT连接断开，中止发送");
      break;
    }
    
    int startPos = i * BASE64_CHUNK_SIZE;
    int chunkLength = min(BASE64_CHUNK_SIZE, totalLength - startPos);
    String chunk = imageBase64.substring(startPos, startPos + chunkLength);
    
    reportImageChunk(chunk.c_str(), i + 1, totalChunks);
    
    for(int j = 0; j < 10; j++) {
      delay(10);
      mqttClient.loop();
    }
  }
  
  // 3. 发送完成标记
  bool completeSuccess = reportImageComplete(totalChunks);
  mqttClient.loop();
  
  esp_camera_fb_return(fb);
  isCapturing = false;
  
  if (completeSuccess) {
    Serial.println("✅ 图片数据上传完成");
    lastActivity = "拍照完成";
  } else {
    Serial.println("❌ 完成标记发送失败");
    lastActivity = "拍照失败";
  }
}

// ==================== 上报开始标记 ====================
bool reportImageStart(int totalChunks, int imageSize) {
  char properties[512];
  char jsonBuf[1024];
  
  sprintf(properties, "\"ImageAction\":\"start\",\"TotalChunks\":%d,\"ImageSize\":%d,\"Timestamp\":%lu,\"Format\":\"jpg\"", 
          totalChunks, imageSize, millis());
  sprintf(jsonBuf, MQTT_BODY_FORMAT, properties);
  
  bool success = mqttClient.publish(MQTT_TOPIC_REPORT, jsonBuf);
  if (success) {
    Serial.println("✅ 开始标记发送成功");
  } else {
    Serial.println("❌ 开始标记发送失败");
    errorCount++;
  }
  return success;
}

// ==================== 上报分片数据 ====================
void reportImageChunk(const char* chunkData, int chunkIndex, int totalChunks) {
  char properties[2048];
  char jsonBuf[3072];
  
  sprintf(properties, "\"ImageAction\":\"chunk\",\"ChunkIndex\":%d,\"TotalChunks\":%d,\"ImageData\":\"%s\"", 
          chunkIndex, totalChunks, chunkData);
  sprintf(jsonBuf, MQTT_BODY_FORMAT, properties);
  
  bool success = mqttClient.publish(MQTT_TOPIC_REPORT, jsonBuf);
  if (!success) {
    Serial.printf("❌ 分片 %d 发送失败\n", chunkIndex);
    errorCount++;
  }
}

// ==================== 上报完成标记 ====================

  } else {

  }

}

// ==================== 系统状态上报（定时上报，不包含AIResult） ====================
bool reportSystemStatus() {
  if (!mqttClient.connected()) {
    Serial.println("❌ MQTT未连接，无法上报状态");
    return false;
  }
  
  if (ESP.getFreeHeap() < 10000) {
    Serial.printf("⚠️ 内存不足: %d，跳过状态上报\n", ESP.getFreeHeap());
    return false;
  }

// ==================== JSON字符串转义函数 ====================
String escapeJSONString(const String& input) {
  String output;
  output.reserve(input.length() + 10);
  
  for (size_t i = 0; i < input.length(); i++) {
    char c = input.charAt(i);
    switch (c) {
      case '"': output += "\\\""; break;
      case '\\': output += "\\\\"; break;
      case '\b': output += "\\b"; break;
      case '\f': output += "\\f"; break;
      case '\n': output += "\\n"; break;
      case '\r': output += "\\r"; break;
      case '\t': output += "\\t"; break;
      default: output += c; break;
    }
  }
  
  return output;
}

def find_nearest_event(events):
    """
    找到离当前时间最近的任务（包括所有未来任务）
    """
    if not events:
        return None
    
    current_time = datetime.now()
    nearest_event = None
    min_time_diff = float('inf')
    
    print(f"🔍 开始查找最近任务，当前时间: {current_time.strftime('%Y-%m-%d %H:%M:%S')}")
    
    for event in events:
        try:
            # 解析事件时间（格式: "01-13 10:30"）
            event_time_str = event['start_time']
            now = datetime.now()
            
            # 解析月-日 时:分
            month_day, time_part = event_time_str.split(' ')
            month, day = map(int, month_day.split('-'))
            hour, minute = map(int, time_part.split(':'))
            
            # 创建日期时间对象（使用当前年份）
            event_time = datetime(now.year, month, day, hour, minute)
            
            # 如果事件时间已经过去（比如去年的事件），则使用下一年的日期
            if event_time < current_time:
                event_time = datetime(now.year + 1, month, day, hour, minute)
            
            # 计算时间差（秒）
            time_diff = (event_time - current_time).total_seconds()
            
            print(f"   📅 事件 '{event['title']}': {event_time.strftime('%Y-%m-%d %H:%M:%S')}, 时间差: {time_diff/60:.1f}分钟")
            
            # 只考虑未来的事件，找到时间差最小的
            if time_diff >= 0 and time_diff < min_time_diff:
                min_time_diff = time_diff
                nearest_event = event.copy()  # 创建副本

    else:
        hours = total_minutes // 60

            return f"{hours}h"

def parse_all_future_events(ical_content):
    """
    解析所有未来事件（不限制时间范围）
    """
    future_events = []
    
    try:
        current_time = datetime.now()
        
        print(f"🕒 当前时间: {current_time.strftime('%Y-%m-%d %H:%M:%S')}")
        print(f"🔍 查找所有未来事件...")
        
        # 解析逻辑
        lines = ical_content.split('\n')
        event_data = {}
        in_event = False
        event_count = 0
        
        for line in lines:
            line = line.strip()
            
            if line == 'BEGIN:VEVENT':
                event_data = {}
                in_event = True
                event_count += 1
            elif line == 'END:VEVENT':
                in_event = False
                # 处理完整的事件
                if event_data:
                    event_info = process_event_for_future(event_data, current_time)
                    if event_info:
                        future_events.append(event_info)
                event_data = {}
            elif in_event and ':' in line:
                # 处理带参数的字段
                parts = line.split(':', 1)
                if len(parts) == 2:
                    key_part, value = parts
                    key = key_part.split(';')[0]  # 提取主字段名
                    event_data[key] = value
        
        print(f"📊 总共解析了 {event_count} 个事件，找到 {len(future_events)} 个未来事件")
        
    except Exception as e:
        print(f"❌ 解析iCal内容异常: {str(e)}")
    
    return future_events

def process_event_for_future(event_data, current_time):
    """
    处理事件数据 - 只筛选未来事件
    """
    try:
        summary = extract_summary(event_data)
            
        # 解析开始时间
        dtstart = event_data.get('DTSTART', '')
        if not dtstart:
            return None
            
        start_time = parse_ical_datetime_fixed(dtstart)
        if not start_time:
            return None
        
        # 如果事件时间已经过去（比如去年的事件），则使用下一年的日期
        if start_time < current_time:
            start_time = datetime(current_time.year + 1, start_time.month, start_time.day, 
                                start_time.hour, start_time.minute, start_time.second)
        
        # 只返回未来事件
        if start_time >= current_time:
            return {
                "title": summary,
                "start_time": start_time.strftime("%m-%d %H:%M"),
                "location": event_data.get('LOCATION', ''),
                "description": event_data.get('DESCRIPTION', '')[:50]
            }
            
    except Exception as e:
        print(f"❌ 处理事件数据异常: {str(e)}")
    
    return None

def extract_summary(event_data):
    """提取事件标题"""
    if 'SUMMARY' in event_data:
        return event_data['SUMMARY'].strip() or '无标题'
    return '无标题'

def parse_ical_datetime_fixed(dt_string):
    """
    修复后的时间解析 - 处理多种iCal时间格式
    """
    try:
        # 处理带时区的格式: 20260113T103000
        if 'T' in dt_string and len(dt_string) >= 15:
            # 移除可能的时区后缀
            dt_clean = dt_string
            if 'Z' in dt_clean:
                dt_clean = dt_clean.replace('Z', '')
            
            # 解析基础格式: YYYYMMDDTHHMMSS
            if len(dt_clean) == 15:
                year = int(dt_clean[0:4])
                month = int(dt_clean[4:6])
                day = int(dt_clean[6:8])
                hour = int(dt_clean[9:11])
                minute = int(dt_clean[11:13])
                second = int(dt_clean[13:15])
                
                return datetime(year, month, day, hour, minute, second)
                
        # 处理全天事件格式: YYYYMMDD
        elif len(dt_string) == 8 and dt_string.isdigit():
            year = int(dt_string[0:4])
            month = int(dt_string[4:6])
            day = int(dt_string[6:8])
            return datetime(year, month, day, 9, 0, 0)  # 默认上午9点
            
    except Exception as e:
        print(f"❌ 解析日期时间异常: {dt_string}, 错误: {e}")
    
    return None

def extract_device_id(event):
    """从事件中提取设备ID"""
    try:
        if isinstance(event, dict) and 'notify_data' in event:

    except Exception as e:

    return None
