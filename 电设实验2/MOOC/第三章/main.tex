\begin{verbatim}
void setup() {

}
void loop() {
  for (int i = 0; i <= 255; i++)
  {
    analogWrite(10, i);
    delay(10);
  }
  delay(50);
  for (int i = 255; i >= 0; i--)

}

\subsection*{3.2 串口控制}

编写以下代码：
\begin{verbatim}
int LedPin = 13;
char ch;

void setup() {
    pinMode(LedPin, OUTPUT);
    Serial.begin(9600);
    Serial.println("Program Start...");
    delay(1000);
}

void loop() {
    if (Serial.available() > 0) {
        ch = Serial.read();
        if (ch == 'A') {
            digitalWrite(LedPin, HIGH);
            Serial.println(ch);
        } else if (ch == 'B') {
            digitalWrite(LedPin, LOW);
            Serial.println(ch);
        } else {
            Serial.println("Error");
        }
    }
}
\end{verbatim}

上传程序，可以观察到：串口先初始化输出"Program Start..."串口输入A时，板载LED灯亮，串口输入B时，板载LED灯灭。

\subsection*{3.3 超声波传感器}

编写以下代码：
\begin{verbatim}
int TrigPin = 2;
int EchoPin = 3;
float cm;

void setup() {
  Serial.begin(9600);
  pinMode(TrigPin,OUTPUT);
  pinMode(TrigPin,INPUT);
}

void loop() {
  digitalWrite(TrigPin,LOW);
  delayMicroseconds(2);
  digitalWrite(TrigPin,HIGH);
  delayMicroseconds(10);
  digitalWrite(TrigPin,LOW);

  cm = pulseIn(EchoPin,HIGH)/58.8;
  cm = (int(cm*100.0))/100.0;

  Serial.print(cm);
  Serial.print("cm");
  Serial.println();
  delay(1000);
}

\end{verbatim}

上传程序，可以在串口监视器中看到每隔1s测量的距离。
